mySQL教程

  • 连接数据库 
                  
  [root@host]# mysql -u root -p
   Enter password:******
  • create RUNOOB; 创建数据库
  • drop RUNOOB; 删除数据库
  • use RUNOOB; 使用数据库
  • 数据库类型
  • SHOW TABLES; 查看表结构
  • desc RUNOOB; 查看数据库内容
  • 插入数据： 
                    
 INSERT INTO runoob
    -> (runoob_title, runoob_author, submission_date)
    -> VALUES
    -> ("php", "involute", '2017-12-26');
  • 读取数据：select * from runoob;
  • 查询数据库：
                      
SELECT column_name,column_name
FROM table_name
[WHERE Clause]
[LIMIT N][ OFFSET M]
  • 修改数据
    
UPDATE table_name SET field1=new-value1, field2=new-value2
[WHERE Clause]
  • 删除数据
    
DELETE FROM table_name [WHERE Clause]

    选择对象删除： delete from students where age<20;

  • 使用LIKE查询关键字的数据：

    SELECT * from runoob_tbl WHERE runoob_author LIKE '%COM';

  • 排序：

    SELECT field1, field2,...fieldN table_name1, table_name2... ORDER BY field1, [field2...] [ASC [DESC]]

  • 分组：
    
SELECT coalesce(runoob_author,'sum'), SUM(runoob_id) as idaccount 
        FROM emp
       GROUNP BY runoob_author  WITH ROLLUP

2. PHP操作mySQL用法

  • 连接数据库 
                  
  $dbhost = 'localhost:3306';  // mysql服务器主机地址
  $dbuser = 'root';            // mysql用户名
  $dbpass = '123456';          // mysql用户名密码
  $conn = mysqli_connect($dbhost, $dbuser, $dbpass);
  if(! $conn )
  {
    die('Could not connect: ' . mysqli_error());
  }
  echo '数据库连接成功！';
  mysqli_close($conn);
  • 创建和删除数据库： 
                  
$dbhost = 'localhost:3306';  // mysql服务器主机地址
$dbuser = 'root';            // mysql用户名
$dbpass = '123456';          // mysql用户名密码
$conn = mysqli_connect($dbhost, $dbuser, $dbpass);
if(! $conn )
{
    die('连接失败: ' . mysqli_error($conn));
}
echo '连接成功
    ';
$sql = 'CREAT DATABASE RUNOOB'; //$sql = 'DROP DATABASE RUNOOB';
$retval = mysqli_query( $conn, $sql );
if(! $retval )
{
    die('创建（删除）数据库失败: ' . mysqli_error($conn));
}
echo "数据库 RUNOOB 创建（删除）成功\n";
mysqli_close($conn);
  • 选择使用数据库：
    
$dbhost = 'localhost:3306';  // mysql服务器主机地址
